import folium from folium import plugins

main_map = folium.Map(location=[4, -72], zoom_start=7)

points = [ {'name': 'Yatacué- Anchicayá', 'coords': [3.57472, -76.87777]}, {'name': 'Estación Zygia', 'coords': [3.44050, -76.650]}, {'name': 'Universidad ICESI', 'coords': [3.34166, -76.52914]} ] for point in points: folium.CircleMarker( location=point['coords'], radius=5, color='red', fill=True, fill_color='red', fill_opacity=1, tooltip=f"Coordenadas: {point['coords'][0]}, {point['coords'][1]}" ).add_to(main_map)

reference_map = folium.Map(location=[4, -72], zoom_start=6)

folium.TileLayer(tiles='Stamen Terrain', overlay=True).add_to(main_map) folium.TileLayer(tiles='CartoDB positron', overlay=True).add_to(reference_map)

folium.plugins.FloatImage('https://raw.githubusercontent.com/SECOORA/static_assets/master/maps/img/rose.png', bottom=5, left=5).add_to(main_map)

minimap = plugins.MiniMap(toggle_display=True) main_map.add_child(minimap)
